// src/app/sitemap.js

import siteConfig from "../data/siteConfig";
import services from "../data/services";
import projects from "../data/projects";
import blogs from "../data/blogs";

/**
 * Next.js App Router sitemap
 * Automatically served as /sitemap.xml
 */
export default async function sitemap() {
  const base =
    typeof siteConfig.siteUrl === "string"
      ? siteConfig.siteUrl.replace(/\/$/, "")
      : "";

  if (!base) return [];

  const pages = ["", "about", "services", "projects", "blog", "contact"];

  const seen = new Set();
  const urls = [];

  // Static pages
  pages.forEach((p) => {
    const url = `${base}/${p}`;
    if (!seen.has(url)) {
      seen.add(url);
      urls.push({
        url,
        lastModified: siteConfig.siteLastUpdated
          ? new Date(siteConfig.siteLastUpdated).toISOString()
          : undefined,
      });
    }
  });

  // Services
  services.forEach((s) => {
    if (!s?.slug) return;
    const url = `${base}/services/${s.slug}`;
    if (!seen.has(url)) {
      seen.add(url);
      urls.push({
        url,
        lastModified: s.lastModified || s.updatedAt
          ? new Date(s.lastModified || s.updatedAt).toISOString()
          : undefined,
      });
    }
  });

  // Projects
  projects.forEach((p) => {
    if (!p?.slug) return;
    const url = `${base}/projects/${p.slug}`;
    if (!seen.has(url)) {
      seen.add(url);
      urls.push({
        url,
        lastModified: p.lastModified || p.updatedAt
          ? new Date(p.lastModified || p.updatedAt).toISOString()
          : undefined,
      });
    }
  });

  // Blogs
  blogs.forEach((b) => {
    if (!b?.slug) return;
    const url = `${base}/blog/${b.slug}`;
    if (!seen.has(url)) {
      seen.add(url);
      urls.push({
        url: url,
        lastModified: b.lastModified || b.updatedAt || b.date
          ? new Date(b.lastModified || b.updatedAt || b.date).toISOString()
          : undefined,
      });
    }
  });

  return urls;
}

﻿"use client";

import { useState } from "react";
import siteConfig from "../data/siteConfig";

/**
 * HOSPITAL APPOINTMENT / INQUIRY FORM
 * Clean medical styling, patient-focused messaging,
 * and professional hospital communication.
 */
export default function LeadForm({ hidden = false }) {
  const [name, setName] = useState("");
  const [phone, setPhone] = useState("");
  const [message, setMessage] = useState("");
  const [status, setStatus] = useState(null);
  const [submitted, setSubmitted] = useState(false);

  async function handleSubmit(e) {
    e.preventDefault();

    if (!name || !phone) {
      setStatus({
        ok: false,
        msg: "Please enter your name and contact number.",
      });
      return;
    }

    const payload = {
      name,
      phone,
      message,
      source: typeof window !== "undefined" ? window.location.pathname : "/",
      date: new Date().toISOString(),
      hospital: siteConfig.companyName,
      city: siteConfig.defaultCity,
    };

    try {
      const res = await fetch("/api/lead", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(payload),
      });

      if (!res.ok) throw new Error();

      setSubmitted(true);
      setStatus(null);
      setName("");
      setPhone("");
      setMessage("");
    } catch {
      setStatus({
        ok: false,
        msg: "Something went wrong. Please try again.",
      });
    }
  }

  /* SUCCESS STATE */
  if (submitted) {
    return (
      <div
        className={`${hidden ? "hidden" : ""} 
          bg-secondary p-10 lg:p-16
          border border-border shadow-2xl
          flex flex-col items-center text-center justify-center min-h-[400px] transition-all duration-500`}
      >
        <div className="w-16 h-16 bg-accent/10 rounded-full flex items-center justify-center mb-6">
          <span className="text-accent text-2xl font-bold">✓</span>
        </div>

        <h4 className="text-2xl font-bold text-text-primary mb-4 tracking-tight">
          Appointment Request Received
        </h4>

        <p className="text-sm text-text-muted leading-relaxed max-w-xs">
          Thank you. Our hospital team will contact you shortly to confirm your
          appointment or assist you with your medical inquiry.
        </p>
      </div>
    );
  }

  /* FORM */
  return (
    <form
      onSubmit={handleSubmit}
      className={`${hidden ? "hidden" : ""} 
        bg-secondary p-8 lg:p-12
        border border-border shadow-2xl transition-all duration-300`}
    >
      {/* Header */}
      <div className="mb-10">
        <div className="flex items-center gap-2 mb-3">
          <span className="w-6 h-[1px] bg-accent" />
          <span className="text-[10px] font-black uppercase tracking-[0.3em] text-accent">
            Patient Support
          </span>
        </div>

        <h4 className="text-2xl text-text-primary font-extrabold tracking-tight">
          Request an Appointment
        </h4>
      </div>

      {/* Fields */}
      <div className="space-y-6">
        {/* Name */}
        <div className="space-y-1.5">
          <label className="text-[10px] font-bold uppercase tracking-widest text-text-muted ml-1">
            Full Name
          </label>
          <input
            value={name}
            onChange={(e) => setName(e.target.value)}
            placeholder="e.g. Rahul Sharma"
            className="
              w-full px-5 py-4 text-sm
              bg-secondary
              text-text-primary
              border border-border
              focus:outline-none focus:border-accent focus:ring-1 focus:ring-accent/20
              transition-all duration-200
            "
          />
        </div>

        {/* Phone */}

/**
 * SEO helpers used across pages to build metadata and JSON-LD.
 * Keep it simple: build title, description, open graph and canonical.
 */

import siteConfig from "../data/siteConfig";

export function buildMeta({ title, description, keywords = [], path = "/", image }) {
  const fullTitle = title ? `${title} | ${siteConfig.companyName}` : siteConfig.seo.defaultTitle;
  const url = new URL(path, siteConfig.siteUrl).toString();

  return {
    title: fullTitle,
    description: description || siteConfig.description,
    keywords: Array.from(new Set([...(keywords || []), ...siteConfig.seo.keywords])).join(", "),
    openGraph: {
      title: fullTitle,
      description: description || siteConfig.description,
      url,
      images: image ? [{ url: new URL(image, siteConfig.siteUrl).toString() }] : [],
    },
    canonical: url,
  };
}

/** Basic JSON-LD for Organization + Website. Additional schema (FAQ, Blog) can be appended in pages. */
export function organizationSchema() {
  return {
    "@context": "https://schema.org",
    "@type": "Organization",
    name: siteConfig.companyName,
    url: siteConfig.siteUrl,
    logo: new URL(siteConfig.logo, siteConfig.siteUrl).toString(),
    contactPoint: [
      {
        "@type": "ContactPoint",
        telephone: siteConfig.phone,
        contactType: "customer service",
        areaServed: "IN",
      },
    ],
  };
}

        {videos.length > 0 ? (
          <div className="grid grid-cols-1 md:grid-cols-2 gap-12">
            {videos.slice(0, 4).map((video) => (
              <VideoEmbed key={video.id} id={video.id} title={video.title} />
            ))}
          </div>
        ) : (
          <div className="max-w-xl text-gray-600">
            <p>
              Health awareness videos, doctor messages, and hospital updates from{" "}
              {siteConfig.companyName} will be added soon.
            </p>
          </div>
        )}
      </div>
    </section>
  );
}
